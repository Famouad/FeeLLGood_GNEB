#===========================================================================
# rev_CMK version number
#===========================================================================
set(PROJECT_MAJOR_VERSION 0)
set(PROJECT_MINOR_VERSION 6)
set(PROJECT_PATCH_VERSION 3)
#===========================================================================

if(POLICY CMP0048)
  cmake_policy(SET CMP0048 NEW) # CMake 3.0.0
endif()

PROJECT (rev_CMK VERSION ${PROJECT_MAJOR_VERSION}.${PROJECT_MINOR_VERSION}.${PROJECT_PATCH_VERSION} )
cmake_minimum_required(VERSION 3.0)

ENABLE_LANGUAGE(CXX)

MESSAGE(STATUS "CXX ${CMAKE_CXX_COMPILER_ID}")

# rev_CMK_VERSION generated by project()
MESSAGE(STATUS "version = ${rev_CMK_VERSION}")

set( CMAKE_CXX_FLAGS "-DNDEBUG -Wall -Wextra -std=c++11 -O3 -march=native" CACHE STRING "C++ compiler flags used during builds." FORCE )

SET(HEADERS rev_CMK.h)

SET(SOURCES rev_CMK.cpp)

ADD_EXECUTABLE(rev_CMK rev_CMK.cpp ${HEADERS} ${SOURCES})
